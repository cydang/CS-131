#!/bin/bash

case "$1" in
    ("" | *[!0-9]*)
        echo 'Error: first parameter is not a positive integer' >&2
        exit 1
esac

if [ "$1" -lt 1 ] || [ "$1" -gt 99 ]; then
    echo 'Error: integer out of range' >&2
    exit 1
fi

echo "valid integer"


if [ -f "$2" ]; then
    echo "$2 file exists"
else
    echo "$2 file does not exist"
fi

max_lines=`wc -l $2 | cut -d' ' -f1`
num_lines=`expr $max_lines \* $1 / 100`
echo "number of lines:"
echo $num_lines

while [ "$num_lines" -gt 0 ]; do
     random_line=`shuf -i 1-$max_lines -n 1`
     sed -n "${random_line}p" $2
     num_lines=`expr $num_lines - 1`
done
